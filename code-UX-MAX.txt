    async function onConfigChange(config) {
      const companyNameEl = document.getElementById('company-name');
      const tokenCurrentEl = document.getElementById('token-current');
      const tokenMaxEl = document.getElementById('token-max');
      const leadsCountEl = document.getElementById('leads-count');
      const messagesCountEl = document.getElementById('messages-count');
      
      if (companyNameEl) companyNameEl.textContent = config.company_name || defaultConfig.company_name;
      if (tokenCurrentEl) tokenCurrentEl.textContent = config.token_current || defaultConfig.token_current;
      if (tokenMaxEl) tokenMaxEl.textContent = config.token_max || defaultConfig.token_max;
      if (leadsCountEl) leadsCountEl.textContent = config.leads_count || defaultConfig.leads_count;
      if (messagesCountEl) messagesCountEl.textContent = config.messages_count || defaultConfig.messages_count;

      const baseFontStack = '-apple-system, BlinkMacSystemFont, "Segoe UI", sans-serif';
      document.body.style.fontFamily = `${customFont}, ${baseFontStack}`;

      document.documentElement.style.setProperty('--base-font-size', `${baseSize}px`);
      
      const pageTitle = document.querySelector('.page-title');

      const kpiLabels = document.querySelectorAll('.kpi-label');

      navItems.forEach(el => el.style.fontSize = `${baseSize}px`);

      const surfaceColor = config.surface_color || defaultConfig.surface_color;
      const textColor = config.text_color || defaultConfig.text_color;
      const primaryAction = config.primary_action || defaultConfig.primary_action;
      const secondaryAction = config.secondary_action || defaultConfig.secondary_action;
      
      document.body.style.background = backgroundColor;
      document.body.style.color = textColor;
      
      document.querySelectorAll('.sidebar, .topbar').forEach(el => {
        el.style.background = `${surfaceColor}f2`;
      });

        el.style.background = `${surfaceColor}e6`;
      });

        el.style.color = textColor;
      });
      
      document.querySelectorAll('.kpi-label, .nav-item').forEach(el => {
        el.style.color = '#64748b';
      });
      
      document.querySelectorAll('.kpi-value, .sidebar-title, .token-text').forEach(el => {
        el.style.color = primaryAction;
      });

        el.style.background = `linear-gradient(135deg, ${primaryAction} 0%, ${secondaryAction} 100%)`;
      });

        el.style.background = `linear-gradient(135deg, ${primaryAction} 0%, ${secondaryAction} 100%)`;
      });
      
      document.querySelectorAll('.chart-bar').forEach(el => {
        el.style.background = `linear-gradient(180deg, ${secondaryAction} 0%, ${primaryAction} 100%)`;
      });
    }
